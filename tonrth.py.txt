import streamlit as st

st.set_page_config(page_title="三中精英怪", page_icon="⚠️")
st.title("⚡ 陈仁钻的考验")

if "step" not in st.session_state:
    st.session_state.step = 0
    st.session_state.game_over = False
    st.session_state.messages = []

for msg in st.session_state.messages:
    with st.chat_message(msg["role"]):
        st.markdown(msg["content"])

def process_input(user_input):
    if user_input == "不是老弟你算个鸡毛啊":
        reply = "陈仁钻：666"
        st.session_state.messages.append({"role": "assistant", "content": reply})
        return reply, False
    elif user_input == "一班的":
        reply1 = "陈仁钻：你可以走了"
        reply2 = "⚠️ 坏结局：你被处分了"
        st.session_state.messages.append({"role": "assistant", "content": reply1})
        st.session_state.messages.append({"role": "assistant", "content": reply2})
        return reply2, True
    elif user_input == "不知道":
        reply = "陈仁钻：小兄弟你不诚哦"
        st.session_state.messages.append({"role": "assistant", "content": reply})
        return reply, False
    elif user_input == "老师好":
        reply = "陈仁钻：神经"
        st.session_state.messages.append({"role": "assistant", "content": reply})
        return reply, False
    elif user_input == "其实我是陈瀚洋":
        reply = "陈仁钻：那还不快点叫爸"
        st.session_state.messages.append({"role": "assistant", "content": reply})
        return reply, False
    else:
        reply1 = "陈仁钻：你这家伙在说什么呢"
        reply2 = "陈仁钻：直接请你吃处分"
        reply3 = "⚠️ 坏结局：你被处分了"
        st.session_state.messages.append({"role": "assistant", "content": reply1})
        st.session_state.messages.append({"role": "assistant", "content": reply2})
        st.session_state.messages.append({"role": "assistant", "content": reply3})
        return reply3, True

if not st.session_state.messages:
    st.session_state.messages.append({"role": "assistant", "content": "陈仁钻：老弟你几班的"})
    st.session_state.messages.append({"role": "assistant", "content": "不好，遇到了三中精英怪"})
    st.rerun()

if not st.session_state.game_over:
    user_input = st.chat_input("我：")
    if user_input:
        st.session_state.messages.append({"role": "user", "content": user_input})
        _, game_end = process_input(user_input)
        if game_end:
            st.session_state.game_over = True
        st.rerun()
else:
    st.info("游戏结束。刷新页面可重新开始。")
